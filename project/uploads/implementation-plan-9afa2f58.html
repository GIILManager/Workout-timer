Circular Timer Ring — SVG Spec

Centre of screen. r = 110, cx = cy = 140, strokeWidth = 10. Background circle: stroke="#1C1C1C". Progress arc: strokeDasharray={circumference} driven by Reanimated useAnimatedProps.

  • SET phase: Progress arc fills clockwise from 0 → target. Colour: #22D46E. Timer counts UP inside. Past target: ring pulses green (Reanimated repeating), timer keeps going.
  • BREAK phase: Progress arc depletes clockwise from full → 0. Colour: #3B82F6. Timer counts DOWN inside. Past 0: arc turns #EF4444, inner text shows +MM:SS in red.
  • AMRAP/Failure: Ring shows static open-arc animation (spinning). No target. Text says "GO" then counts up. DONE button is the only control.
  • TIMED (Plank): Same as SET phase but with fixed setDuration=60.

Alert Behaviour

  • When SET timer reaches target: play beep.mp3 (2 beeps) + expo-haptics heavy impact × 2, then silence. Timer continues counting up. DONE button still visible.
  • When BREAK timer reaches 0: play beep.mp3 (3 beeps) + expo-haptics heavy impact × 3, then silence. Timer goes into overtime (red). START SET still visible.
  • At 90 min elapsed: Elapsed clock badge turns amber. Subtle haptic. No interruption.
  • At 120 min elapsed: Full-screen overlay warning, 1 large dismiss button. Haptic + longer tone.
  • User can stop sound/vibration at any time by tapping the DONE or START SET button (which records the tap and advances state anyway).

Thumb-Zone Layout Rules

  • Primary action button (DONE / START NEXT SET): bottom 20% of screen, full width minus 24px padding, height 64px, border-radius 16px.
  • Secondary action (SKIP BREAK): directly above primary, 48px height, outline style.
  • Abandon workout button (✕): top-left, 44×44px tap target.
  • Exercise name + set counter: middle zone (informational, not interactive).
  • Circular ring: fills remaining vertical space between header and buttons.

1.3 — Timing utilities & learning algorithmLOGIC

Write src/utils/timing.ts. This is the heart of the adaptive timing system.

import { ExerciseType, TimingRecord, UserSettings } from '@/types';

export const DEFAULT_SETTINGS: UserSettings = {
  tier1SetDuration: 60,
  tier1BreakDuration: 150,    // 2:30
  standardSetDuration: 40,
  standardBreakDuration: 90,  // 1:30
  timedBreakDuration: 60,     // 1:00
  targetWorkoutMinutes: 90,
  warningWorkoutMinutes: 120,
  minSessionsForAdaptation: 3,
};

export interface TimingPrediction {
  setDuration: number | null; // null = AMRAP
  breakDuration: number;
}

export function getDefaultTiming(
  type: ExerciseType,
  settings: UserSettings
): TimingPrediction {
  switch (type) {
    case 'TIER1':
      return { setDuration: settings.tier1SetDuration, breakDuration: settings.tier1BreakDuration };
    case 'TIMED':
      return { setDuration: 60, breakDuration: settings.timedBreakDuration };
    case 'AMRAP':
      return { setDuration: null, breakDuration: settings.standardBreakDuration };
    default:
      return { setDuration: settings.standardSetDuration, breakDuration: settings.standardBreakDuration };
  }
}

// Rolling average of last N records for this exercise
const ROLLING_WINDOW = 15;

export function getAdaptedTiming(
  exerciseId: string,
  exerciseType: ExerciseType,
  allRecords: TimingRecord[],
  settings: UserSettings
): TimingPrediction {
  const defaults = getDefaultTiming(exerciseType, settings);

  // Count distinct sessions for this exercise
  const relevant = allRecords.filter(r => r.exerciseId === exerciseId);
  const distinctSessions = new Set(relevant.map(r => r.sessionId)).size;

  if (distinctSessions < settings.minSessionsForAdaptation) {
    return defaults; // not enough history yet
  }

  const window = relevant.slice(-ROLLING_WINDOW);

  // Adapt setDuration only for types that have one
  const validSetRecords = window.filter(r => r.setDuration !== null);
  const adaptedSet = validSetRecords.length > 0
    ? Math.round(validSetRecords.reduce((a, r) => a + r.setDuration!, 0) / validSetRecords.length)
    : defaults.setDuration;

  const adaptedBreak = Math.round(
    window.reduce((a, r) => a + r.breakDuration, 0) / window.length
  );

  return {
    setDuration: exerciseType === 'AMRAP' ? null : adaptedSet,
    breakDuration: Math.max(30, adaptedBreak), // floor: 30s break minimum
  };
}

export function estimateTotalDuration(
  workout: { exercises: Array<{ type: ExerciseType; sets: number; id: string }> },
  allRecords: TimingRecord[],
  settings: UserSettings
): number {
  let total = 0;
  for (const ex of workout.exercises) {
    if (ex.type === 'CARDIO') continue;
    const timing = getAdaptedTiming(ex.id, ex.type, allRecords, settings);
    const setTime = timing.setDuration ?? 45; // estimate 45s for AMRAP
    total += ex.sets * (setTime + timing.breakDuration);
  }
  return Math.round(total / 60); // return minutes
}

export function formatTime(seconds: number): string {
  const absSeconds = Math.abs(seconds);
  const m = Math.floor(absSeconds / 60);
  const s = absSeconds % 60;
  const sign = seconds < 0 ? '-' : '';
  return `${sign}${m}:${s.toString().padStart(2, '0')}`;
}

export function formatElapsed(seconds: number): string {
  const h = Math.floor(seconds / 3600);
  const m = Math.floor((seconds % 3600) / 60);
  const s = seconds % 60;
  if (h > 0) return `${h}:${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
  return `${m}:${s.toString().padStart(2, '0')}`;
}

import { create } from 'zustand';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { TimingRecord, WorkoutSession, UserSettings } from '@/types';
import { DEFAULT_SETTINGS } from '@/utils/timing';

const KEYS = {
  timing:   'timing_history',
  sessions: 'session_history',
  settings: 'user_settings',
};

interface HistoryStore {
  timingRecords: TimingRecord[];
  sessions: WorkoutSession[];
  settings: UserSettings;
  hydrated: boolean;

  hydrate: () => Promise;
  addTimingRecord: (record: TimingRecord) => Promise;
  saveSession: (session: WorkoutSession) => Promise;
  updateSettings: (patch: Partial<UserSettings>) => Promise;
  resetLearning: () => Promise;
}

export const useHistoryStore = create<HistoryStore>((set, get) => ({
  timingRecords: [],
  sessions: [],
  settings: DEFAULT_SETTINGS,
  hydrated: false,

  hydrate: async () => {
    try {
      const [timingRaw, sessionsRaw, settingsRaw] = await Promise.all([
        AsyncStorage.getItem(KEYS.timing),
        AsyncStorage.getItem(KEYS.sessions),
        AsyncStorage.getItem(KEYS.settings),
      ]);
      set({
        timingRecords: timingRaw ? JSON.parse(timingRaw) : [],
        sessions:      sessionsRaw ? JSON.parse(sessionsRaw) : [],
        settings:      settingsRaw ? { ...DEFAULT_SETTINGS, ...JSON.parse(settingsRaw) } : DEFAULT_SETTINGS,
        hydrated: true,
      });
    } catch (e) {
      set({ hydrated: true });
    }
  },

  addTimingRecord: async (record) => {
    const next = [...get().timingRecords, record];
    set({ timingRecords: next });
    await AsyncStorage.setItem(KEYS.timing, JSON.stringify(next));
  },

  saveSession: async (session) => {
    const next = [session, ...get().sessions].slice(0, 100); // keep last 100
    set({ sessions: next });
    await AsyncStorage.setItem(KEYS.sessions, JSON.stringify(next));
  },

  updateSettings: async (patch) => {
    const next = { ...get().settings, ...patch };
    set({ settings: next });
    await AsyncStorage.setItem(KEYS.settings, JSON.stringify(next));
  },

  resetLearning: async () => {
    set({ timingRecords: [] });
    await AsyncStorage.removeItem(KEYS.timing);
  },
}));

Create src/utils/alertService.ts. Handles sound + haptics with the 2-second auto-stop behaviour.

import { Audio } from 'expo-av';
import * as Haptics from 'expo-haptics';

let soundObject: Audio.Sound | null = null;
let alertTimeout: ReturnType<typeof setTimeout> | null = null;

async function loadSound() {
  if (soundObject) return;
  const { sound } = await Audio.Sound.createAsync(
    require('../../assets/sounds/beep.mp3')
  );
  soundObject = sound;
}

// Fire alert: sound + haptics, then stop after 2 seconds
export async function fireAlert(beeps: number = 2) {
  // Clear any previous alert
  if (alertTimeout) {
    clearTimeout(alertTimeout);
    alertTimeout = null;
  }

  try {
    await Audio.setAudioModeAsync({
      playsInSilentModeIOS: true,
      staysActiveInBackground: true,
    });

    await loadSound();

    // Play beeps
    for (let i = 0; i < beeps; i++) {
      await soundObject!.setPositionAsync(0);
      await soundObject!.playAsync();
      // Haptic with each beep
      await Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Heavy);
      await new Promise(res => setTimeout(res, 400));
    }

    // Auto-stop after 2 seconds total
    alertTimeout = setTimeout(async () => {
      await soundObject?.stopAsync();
    }, 2000);
  } catch (e) {
    console.warn('Alert service error:', e);
  }
}

export async function stopAlert() {
  if (alertTimeout) {
    clearTimeout(alertTimeout);
    alertTimeout = null;
  }
  await soundObject?.stopAsync();
}

// Different alert intensities
export const Alerts = {
  setComplete: () => fireAlert(2),  // 2 beeps when set target reached
  breakComplete: () => fireAlert(3), // 3 beeps when break ends
  workoutWarning: () => fireAlert(5), // 5 beeps at 2hr warning
};

Create src/hooks/useWorkoutDuration.ts:

import { useState, useEffect, useRef } from 'react';
import { useWorkoutStore } from '@/store/workoutStore';
import { useHistoryStore } from '@/store/historyStore';
import { Alerts } from '@/utils/alertService';

export function useWorkoutDuration() {
  const sessionStartedAt = useWorkoutStore(s => s.sessionStartedAt);
  const warningDismissed = useWorkoutStore(s => s.warningDismissed);
  const dismissWarning = useWorkoutStore(s => s.dismissWarning);
  const settings = useHistoryStore(s => s.settings);

  const [elapsedSeconds, setElapsedSeconds] = useState(0);
  const [showWarning, setShowWarning] = useState(false);
  const warningFiredRef = useRef(false);

  useEffect(() => {
    if (!sessionStartedAt) { setElapsedSeconds(0); return; }

    const interval = setInterval(() => {
      const elapsed = Math.floor((Date.now() - sessionStartedAt) / 1000);
      setElapsedSeconds(elapsed);

      const elapsedMinutes = elapsed / 60;
      if (
        elapsedMinutes >= settings.warningWorkoutMinutes &&
        !warningFiredRef.current &&
        !warningDismissed
      ) {
        warningFiredRef.current = true;
        setShowWarning(true);
        Alerts.workoutWarning();
      }
    }, 1000);

    return () => clearInterval(interval);
  }, [sessionStartedAt, settings.warningWorkoutMinutes, warningDismissed]);

  const dismiss = () => {
    setShowWarning(false);
    dismissWarning();
  };

  const targetSeconds = settings.targetWorkoutMinutes * 60;
  const warningSeconds = settings.warningWorkoutMinutes * 60;
  const progressToTarget = Math.min(elapsedSeconds / targetSeconds, 1);
  const isNearTarget = elapsedSeconds >= targetSeconds * 0.9; // 90% of target
  const isPastTarget = elapsedSeconds >= targetSeconds;

  return { elapsedSeconds, showWarning, dismiss, progressToTarget, isNearTarget, isPastTarget };
}
✓ Acceptance
At 90 min: elapsed clock turns amber. At 120 min: showWarning becomes true. Warning fires only once even if component re-renders. Dismissing sets showWarning=false and does not re-trigger.

#!/usr/bin/env python3
"""Patches android/app/build.gradle to add release signing config."""
import re, pathlib

gradle_path = pathlib.Path('android/app/build.gradle')
content = gradle_path.read_text()

signing_config = """
    signingConfigs {
        release {
            storeFile file('workout-timer.keystore')
            storePassword System.getProperty('ANDROID_STORE_PASSWORD', '') ?: new File('../key.properties').with { f -> f.exists() ? (new java.util.Properties().tap { load(f.newInputStream()) })['storePassword'] : '' }
            keyAlias System.getProperty('ANDROID_KEY_ALIAS', '') ?: new File('../key.properties').with { f -> f.exists() ? (new java.util.Properties().tap { load(f.newInputStream()) })['keyAlias'] : '' }
            keyPassword System.getProperty('ANDROID_KEY_PASSWORD', '') ?: new File('../key.properties').with { f -> f.exists() ? (new java.util.Properties().tap { load(f.newInputStream()) })['keyPassword'] : '' }
        }
    }
"""

if 'signingConfigs' not in content:
    content = content.replace('buildTypes {', signing_config + '\n    buildTypes {')

# Update release buildType to use signing config
content = re.sub(
    r'(release \{[^}]*)(minifyEnabled)',
    r'\1signingConfig signingConfigs.release\n            \2',
    content
)

gradle_path.write_text(content)
print("build.gradle patched successfully")
